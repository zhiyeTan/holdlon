
/* =============================================================== */
table的几种设置方法：

V5.0+：
->table('admin')
->table(array('admin', 'user'))

V5.4+：
->table('admin')
->table(['admin', 'user'])



/* =============================================================== */
field的几种设置方法：

V5.0+：
->field('id')
->field(array('account', 'password'))
->field(array('account'=>'count'))
->field(array('account'=>'a', 'id'=>'max'))
->field(array('*'=>'count', 'id'=>'count'))
->field(array(array('account','count','a')))
->field(array(array('account','','a'), array('*','count','b')))

V5.4+：
->field('id')
->field(['account', 'password'])
->field(['account'=>'count'])
->field(['account'=>'a', 'id'=>'max'])
->field(['*'=>'count', 'id'=>'count'])
->field([['account','count','a']]]
->field([['account','','a'], ['*','count','b']])



/* =============================================================== */
from的几种设置方法：

V5.0+：
->from('admin')
->from('admin', 'a')



/* =============================================================== */
join的几种设置方法：

V5.0+：
->join(array('inner', 'user', 'u', array('u.uid', 'a.uid')))
->join(array('left', 'user', 'u', array(array('u.uid', 'a.uid'), array('u.uid', 'b.uid'))))

V5.4+：
->join(['inner', 'user', 'u', ['u.uid', 'a.uid']])
->join(['left', 'user', 'u', [['u.uid', 'a.uid'], ['u.uid', 'b.uid']]])



/* =============================================================== */
where的几种设置方法：

V5.0+：
->where(array('uid', '=', 1))
->where(array(array('uid', '=', 1), array('id', '>=', 12)))
->where(array(array('uid', '=', 1, '&&'), array('id', '>=', 12)))

V5.4+：
->where(['uid', '=', 1])
->where([['uid', '=', 1], ['id', '>=', 12]])
->where([['uid', '=', 1, '&&'], ['id', '>=', 12]])



/* =============================================================== */
group的几种设置方法：

V5.0+：
->group('age')
->group(array('age', 'familyName'))

V5.4+：
->group('age')
->group(['age', 'familyName'])



/* =============================================================== */
group的几种设置方法：

V5.0+：
->having(array('id', '>', 0))
->having(array(array('id', '>', 0, '||'), array('sum(id)', '>', 1)))
->having(array(array('sum(id)', '>', 1, 'and'), array('count(*)', '>', 1)))

V5.4+：
->having(['id', '>', 0])
->having([['id', '>', 0, '||'], ['sum(id)', '>', 1]])
->having([['sum(id)', '>', 1, 'and'], ['count(*)', '>', 1]])



/* =============================================================== */
order的几种设置方法：

V5.0+：
->order('age')
->order(array('age', 'sort'))
->order(array('age'=>'asc', 'sort'=>'desc'))

V5.4+：
->order('age')
->order(['age', 'sort'])
->order(['age'=>'asc', 'sort'=>'desc'])



/* =============================================================== */
limit的几种设置方法：

V5.0+：
->limit(5)
->limit(0, 5)



/* =============================================================== */
data的几种设置方法：

V5.0+：
->data(1)
->data(array(1, 'admin', 'passwordkey'))
->data(array(array(1, 'admin', 'passwordkey'), array(2, 'admin2', 'passwordkey2')))

V5.4+：
->data(1)
->data([1, 'admin', 'passwordkey'])
->data([[1, 'admin', 'passwordkey'], [2, 'admin2', 'passwordkey2']]]
